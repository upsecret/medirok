/**
 * 청담 피부과 지역 가이드 매거진 등록 시드 — 멱등(upsert)
 * 실행: npm run seed:magazine-cheongdam
 * 출처: dodskin.com 공식 + medirok seed-dod.ts 검증분. 미검증 수치 미기입.
 */
import { getPayload } from "payload";
import config from "@payload-config";

type AnyData = Record<string, unknown>;

async function upsertMagazine(
  payload: Awaited<ReturnType<typeof getPayload>>,
  slug: string,
  data: AnyData,
) {
  const existing = await payload.find({
    collection: "magazines",
    where: { slug: { equals: slug } },
    limit: 1,
  });
  if (existing.docs.length > 0) {
    await payload.update({ collection: "magazines", id: existing.docs[0].id, data });
    console.log(`  ✓ update  magazines/${slug}`);
  } else {
    await payload.create({ collection: "magazines", data });
    console.log(`  ✓ create  magazines/${slug}`);
  }
}
